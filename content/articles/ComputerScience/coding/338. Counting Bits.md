class Trie:
    strArr = []
    def __init__(self):
				self.strArr = []
        return None

    def insert(self, word: str) -> None:
        self.strArr.append(word)
        return None
        

    def search(self, word: str) -> bool:
        for x in self.strArr:
            if x == word:
                return True
        return False 
        

    def startsWith(self, prefix: str) -> bool:
        for x in self.strArr:
            if x.startswith(prefix):
                return True
                
        return False
