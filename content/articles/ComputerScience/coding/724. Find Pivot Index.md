class Solution:
    def pivotIndex(self, nums: List[int]) -> int:
        for idx in range(len(nums)):
            lSum = sum(nums[:idx])
            rSum = sum(nums[idx+1:])
            if lSum == rSum:
                return idx

        return -1
```

class Solution:
    def pivotIndex(self, nums: List[int]) -> int:
        left = 0
        right = sum(nums)
        for i, n in enumerate(nums):
            right -= n
            if left == right:
                return i
            left += n
        return -1
```
